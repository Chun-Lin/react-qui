### Row

```js
<div>
  <CheckboxGroup
    options={[
      { label: 'SUB', value: 'sub' },
      { label: 'MNR', value: 'mnr' },
      { label: 'ICX', value: 'icx' },
    ]}
    span={30}
    theme="light"
    direction="row"
  />
</div>
```

### Column

```js
<div>
  <CheckboxGroup
    options={[
      { label: 'ADA', value: 'ada', checked: true },
      { label: 'NEO', value: 'neo', checked: true },
      { label: 'EOS', value: 'eos', checked: true },
    ]}
    span={30}
    theme="light"
    direction="col"
    onChange={e => {
      console.log(`checked: ${e.target.checked}, value: ${e.target.value}`)
    }}
  />
</div>
```
